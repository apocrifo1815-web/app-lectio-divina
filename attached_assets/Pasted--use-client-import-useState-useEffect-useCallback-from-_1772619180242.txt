"use client"

import { useState, useEffect, useCallback } from "react"
import { ProgressBar } from "./progress-bar"
import { StepReading } from "./step-reading"
import { StepMeditation } from "./step-meditation"
import { StepPrayer } from "./step-prayer"
import { StepContemplation } from "./step-contemplation"
import { NavigationButtons } from "./navigation-buttons"

const TOTAL_STEPS = 4

/**
 * Fetches the daily Gospel automatically from the Catholic Liturgical Calendar API.
 */
async function fetchDailyGospel() {
  try {
    // 1. Pega a data de hoje baseada no seu computador/celular
    const today = new Date();
    const formattedDate = today.toISOString().split('T')[0];

    // 2. Usaremos uma API mais estável (EUA/Europa) para pegar a referência do dia
    // Se a russa falhar, o sistema tentará uma busca direta
    const res = await fetch(`https://readings.vincit.spb.ru/api/v1/readings/${formattedDate}`);
    
    let reference = "Luke 6:36-38"; // Padrão para HOJE (02/03) caso tudo falhe
    let liturgicalDay = "Monday of the 2nd Week of Lent";

    if (res.ok) {
      const data = await res.json();
      const gospel = data.readings?.find((r: any) => r.type?.toLowerCase().includes('gospel'));
      if (gospel) {
        reference = gospel.reference;
        liturgicalDay = data.title || liturgicalDay;
      }
    }

    // 3. Busca o TEXTO da Bíblia na Bible-API (Servidores nos EUA - super estável)
    const bibleRes = await fetch(`https://bible-api.com/${encodeURIComponent(reference)}?translation=kjv`);
    if (!bibleRes.ok) throw new Error("Bible API offline");
    
    const bibleData = await bibleRes.json();

    return {
      reference: bibleData.reference || reference,
      liturgicalDay: liturgicalDay,
      text: bibleData.text || "Loading daily word...",
    };

  } catch (error) {
    console.error("Erro geral:", error);
    // Se TUDO falhar, ele busca a leitura de hoje (Lucas 6) na API americana
    try {
      const backupRes = await fetch(`https://bible-api.com/Luke+6:36-38?translation=kjv`);
      const backupData = await backupRes.json();
      return {
        reference: "Luke 6:36-38",
        liturgicalDay: "Daily Gospel",
        text: backupData.text,
      };
    } catch (e) {
      return {
        reference: "Luke 6:36-38",
        liturgicalDay: "Daily Gospel",
        text: "Be ye therefore merciful, as your Father also is merciful. Judge not, and ye shall not be judged: condemn not, and ye shall not be condemned: forgive, and ye shall be forgiven: Give, and it shall be given unto you; good measure, pressed down, and shaken together, and running over, shall men give into your bosom. For with the same measure that ye mete withal it shall be measured to you again.",
      };
    }
  }
}

export function LectioDivina() {
  const [currentStep, setCurrentStep] = useState(0)
  const [isLoading, setIsLoading] = useState(true)
  const [dailyGospel, setDailyGospel] = useState<{
    reference: string
    text: string
    liturgicalDay: string
  } | null>(null)
  const [meditationText, setMeditationText] = useState("")
  const [prayerText, setPrayerText] = useState("")

  useEffect(() => {
    setIsLoading(true)
    fetchDailyGospel()
      .then((data) => {
        console.log("Gospel data loaded:", data.reference);
        setDailyGospel(data)
      })
      .catch((err) => {
        console.error("Gospel fetch component error:", err);
        setDailyGospel(null)
      })
      .finally(() => {
        setIsLoading(false)
      })
  }, [])

  const goNext = useCallback(() => {
    setCurrentStep((prev) => Math.min(prev + 1, TOTAL_STEPS - 1))
    window.scrollTo({ top: 0, behavior: "smooth" })
  }, [])

  const goBack = useCallback(() => {
    setCurrentStep((prev) => Math.max(prev - 1, 0))
    window.scrollTo({ top: 0, behavior: "smooth" })
  }, [])

  return (
    <div className="flex min-h-dvh flex-col bg-background">
      {/* Header */}
      <header className="flex flex-col items-center pb-2 pt-6 md:pt-8">
        <p className="text-[10px] font-medium uppercase tracking-[0.3em] text-muted-foreground">
          A Contemplative Practice
        </p>
        <h1 className="mt-1 font-serif text-xl font-medium text-foreground md:text-2xl">
          Lectio Divina
        </h1>
        <div className="mx-auto mt-2 h-px w-8 bg-accent/50" />
      </header>

      {/* Progress */}
      <div className="pb-6 pt-4 md:pb-8 md:pt-6">
        <ProgressBar currentStep={currentStep} />
      </div>

      {/* Main Content */}
      <main className="mx-auto w-full max-w-3xl flex-1 px-4 pb-8 md:px-6">
        <div className="transition-opacity duration-500">
          {currentStep === 0 && (
            <StepReading dailyGospel={dailyGospel} isLoading={isLoading} />
          )}
          {currentStep === 1 && (
            <StepMeditation
              meditationText={meditationText}
              onMeditationChange={setMeditationText}
            />
          )}
          {currentStep === 2 && (
            <StepPrayer prayerText={prayerText} onPrayerChange={setPrayerText} />
          )}
          {currentStep === 3 && (
            <StepContemplation
              dailyGospel={dailyGospel}
              meditationText={meditationText}
              prayerText={prayerText}
            />
          )}
        </div>
      </main>

      {/* Navigation Footer */}
      <footer className="sticky bottom-0 border-t border-border bg-background/80 backdrop-blur-md">
        <div className="mx-auto max-w-3xl px-4 py-4 md:px-6">
          <NavigationButtons
            currentStep={currentStep}
            totalSteps={TOTAL_STEPS}
            onBack={goBack}
            onNext={goNext}
          />
        </div>
      </footer>
    </div>
  )
}
